- **표시 위치 2:** 프로필 카드의 이름 아래에 인라인 뱃지 형태로 표시

```js
// 예시 로직
function getStreak() {
  const dates = myRecords.map(r => r.date).sort().reverse();
  let streak = 0, check = new Date();
  for (const d of dates) {
    if (d === check.toISOString().slice(0,10)) { streak++; check.setDate(check.getDate()-1); }
    else break;
  }
  return streak;
}
```

function toggleDark() {
  const isDark = document.body.classList.toggle('dark');
  document.getElementById('darkToggleBtn').textContent = isDark ? '☀️' : '🌙';
  localStorage.setItem('darkMode', isDark);
}
// 페이지 로드 시 복원
if (localStorage.getItem('darkMode') === 'true') toggleDark();
